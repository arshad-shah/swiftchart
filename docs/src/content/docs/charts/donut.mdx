    import { PieChart } from '@arshad-shah/swift-chart';

    const chart = new PieChart('#chart', {
      theme: 'midnight',
      donut: true,
      donutWidth: 0.55,
    });
    chart.setData(traffic, { labelField: 'source', valueField: 'visits' });
    ```
  </TabItem>

      theme="midnight"
      donutWidth={0.55}
      height={340}
    />
    ```
  </TabItem>

## Common options

### `donutWidth`

The cutout ratio (0–1). Lower values produce a thicker ring.

<ChartPreview client:visible kind="donut" data={traffic} mapping={{ labelField: 'source', valueField: 'visits' }} options={{ theme: 'arctic', donutWidth: 0.3 }} height={260} />
